讓馬達動起來吧！#

備註

這邊用Kraken跟NEO的程式來舉例

重要

因為省略掉import的部分，所以直接複製貼上動不了是正常的

Defintion#
public SparkMax neo = new SparkMax(10, MotorType.kBrushless);
public TalonFX kraken = new TalonFX(11, new CANBus()); //new CANBus會指到RIO的bus

注意

這邊沒做configuration, 關於configuration之後會再做一篇文章做詳細說明（老高

最簡單讓他動起來的方式

SimpleRun#
neo.set(1);
kraken.set(1);

這個方式是讓馬達用開還控制的方式到指定的轉速(趴數，0~1)

不過Kraken更常見的是用這種方式

KrakenRun#
public DutyCycleOut request = new DutyCycleOut();

//---------- (已省略)

motor.setControl(request.withOutput(1));

這邊會這樣寫別有原因，之後就會懂了

讓我們拉回來，讓我們看看如果整合到系統之後會變成什麼樣子

 1public class Drivetrain implements Subsystem{
 2    public TalonFX FrontLeft, FrontRight, BackLeft, BackRight;
 3    public TalonFXConfiguration FLConfig, FRConfig, BLConfig, BRConfig;
 4    public DutyCycleOut output;
 5    public Follower follower;
 6    public DifferentialDriveKinematics kinematics;
 7    public final Distance WheelCirc = Inches.of(6).times(Math.PI);
 8    public final double GearRatio = 10.71;
 9    public final double MaxVelocity = 100/10.71*WheelCirc.in(Meters);
10
11    public Drivetrain(){
12        FrontLeft = new TalonFX(0);
13        FrontRight = new TalonFX(1);
14        BackLeft = new TalonFX(2);
15        BackRight = new TalonFX(3);
16
17        FLConfig = new TalonFXConfiguration()
18            .withMotorOutput(new MotorOutputConfigs()
19                .withNeutralMode(NeutralModeValue.Brake)
20                .withInverted(InvertedValue.Clockwise_Positive))
21            .withFeedback(new FeedbackConfigs()
22                .withSensorToMechanismRatio(GearRatio));
23
24        FRConfig = new TalonFXConfiguration()
25            .withMotorOutput(new MotorOutputConfigs()
26                .withNeutralMode(NeutralModeValue.Brake)
27                .withInverted(InvertedValue.CounterClockwise_Positive))
28            .withFeedback(new FeedbackConfigs()
29                .withSensorToMechanismRatio(GearRatio));
30
31        BLConfig = new TalonFXConfiguration()
32            .withMotorOutput(new MotorOutputConfigs()
33                .withNeutralMode(NeutralModeValue.Brake)
34                .withInverted(InvertedValue.Clockwise_Positive))
35            .withFeedback(new FeedbackConfigs()
36                .withSensorToMechanismRatio(GearRatio));
37
38        BRConfig = new TalonFXConfiguration()
39            .withMotorOutput(new MotorOutputConfigs()
40                .withNeutralMode(NeutralModeValue.Brake)
41                .withInverted(InvertedValue.CounterClockwise_Positive))
42            .withFeedback(new FeedbackConfigs()
43                .withSensorToMechanismRatio(GearRatio));
44
45        FrontLeft.getConfigurator().apply(FLConfig);
46        FrontRight.getConfigurator().apply(FRConfig);
47        BackLeft.getConfigurator().apply(BLConfig);
48        BackRight.getConfigurator().apply(BRConfig);
49
50        output = new DutyCycleOut(0);
51        follower = new Follower(0, MotorAlignmentValue.Aligned);
52        kinematics = new DifferentialDriveKinematics(0.0);
53    }
54
55    public Command drive(Supplier<ChassisSpeeds> speeds){
56        return run( () -> {
57            DifferentialDriveWheelSpeeds wheelspeed = kinematics.toWheelSpeeds(speeds.get());
58            FrontLeft.setControl(output.withOutput(wheelspeed.leftMetersPerSecond/MaxVelocity));
59            FrontRight.setControl(output.withOutput(wheelspeed.rightMetersPerSecond/MaxVelocity));
60            BackLeft.setControl(follower.withLeaderID(FrontLeft.getDeviceID()));
61            BackRight.setControl(follower.withLeaderID(FrontRight.getDeviceID()));
62        });
63    }
64}

SwerveMod.java#
 1public class SwerveMod {
 2    public TalonFX DriveMotor, SteerMotor;
 3    public CANcoder encoder;
 4    public MotionMagicVelocityTorqueCurrentFOC DriveRequest;
 5    public MotionMagicExpoTorqueCurrentFOC SteerRequest;
 6
 7    private TalonFXConfiguration DriveConfig, SteerConfig;
 8    private CANcoderConfiguration EncoderConfig;
 9
10    public SwerveMod(int DriveID, int SteerID, int EncoderID, Angle offset, boolean isLeftSide){
11        DriveMotor = new TalonFX(DriveID);
12        SteerMotor = new TalonFX(SteerID);
13        encoder = new CANcoder(EncoderID);
14
15        DriveRequest = new MotionMagicVelocityTorqueCurrentFOC(0);
16        SteerRequest = new MotionMagicExpoTorqueCurrentFOC(0);
17
18        DriveConfig = new TalonFXConfiguration();
19        SteerConfig = new TalonFXConfiguration();
20        EncoderConfig = new CANcoderConfiguration();
21
22        DriveConfig.MotorOutput
23            .withInverted(isLeftSide ? InvertedValue.Clockwise_Positive : InvertedValue.CounterClockwise_Positive)
24            .withNeutralMode(NeutralModeValue.Brake);
25        DriveConfig.Feedback
26            .withSensorToMechanismRatio(Constants.DriveGearRatio);
27        DriveConfig.Slot0
28            .withKP(0).withKD(0)
29            .withStaticFeedforwardSign(StaticFeedforwardSignValue.UseVelocitySign);
30        DriveConfig.CurrentLimits
31            .withStatorCurrentLimit(Constants.SlipCurrent)
32            .withStatorCurrentLimitEnable(true);
33        DriveConfig.MotionMagic
34            .withMotionMagicAcceleration(0);
35
36        SteerConfig.MotorOutput
37            .withInverted(InvertedValue.Clockwise_Positive)
38            .withNeutralMode(NeutralModeValue.Brake);
39        DriveConfig.Feedback
40            .withFusedCANcoder(encoder)
41            .withRotorToSensorRatio(Constants.SteerGearRatio);
42        DriveConfig.Slot0
43            .withKP(0).withKD(0)
44            .withStaticFeedforwardSign(StaticFeedforwardSignValue.UseVelocitySign);
45        DriveConfig.CurrentLimits
46            .withStatorCurrentLimit(Amps.of(60))
47            .withStatorCurrentLimitEnable(true);
48        DriveConfig.MotionMagic
49            .withMotionMagicExpo_kV(0)
50            .withMotionMagicExpo_kA(0);
51
52        EncoderConfig.MagnetSensor
53            .withMagnetOffset(offset);
54
55        DriveMotor.getConfigurator().apply(DriveConfig);
56        SteerMotor.getConfigurator().apply(SteerConfig);
57        encoder.getConfigurator().apply(EncoderConfig);
58    }
59
60    public SwerveModulePosition getPosition(){
61        return new SwerveModulePosition(
62            DriveMotor.getPosition().getValueAsDouble() * Constants.WheelRadius.times(2*Math.PI).in(Meters),
63            new Rotation2d(SteerMotor.getPosition().getValue())
64        );
65    }
66
67    public SwerveModuleState getState(){
68        return new SwerveModuleState(
69            DriveMotor.getVelocity().getValueAsDouble() * Constants.WheelRadius.times(2*Math.PI).in(Meters),
70            new Rotation2d(SteerMotor.getPosition().getValue())
71        );
72    }
73
74    public void setState(SwerveModuleState state){
75        state.optimize(getState().angle);
76
77        DriveMotor.setControl(DriveRequest.withVelocity(state.speedMetersPerSecond/Constants.WheelRadius.times(2*Math.PI).in(Meters)));
78        SteerMotor.setControl(SteerRequest.withPosition(state.angle.getMeasure()));
79    }
80}

 1public class Drivetrain implements Subsystem{
 2    public List<SwerveMod> modules;
 3    public Pigeon2 gyro;
 4    public SwerveDriveKinematics kinematics;
 5
 6    public Drivetrain(){
 7        modules = List.of(
 8            new SwerveMod(FrontLeft.DriveID, FrontLeft.SteerID, FrontLeft.EncoderID, FrontLeft.Offset, true),
 9            new SwerveMod(FrontRight.DriveID, FrontRight.SteerID, FrontRight.EncoderID, FrontRight.Offset, false),
10            new SwerveMod(BackLeft.DriveID, BackLeft.SteerID, BackLeft.EncoderID, BackLeft.Offset, true),
11            new SwerveMod(BackRight.DriveID, BackRight.SteerID, BackRight.EncoderID, BackRight.Offset, false)
12        );
13
14        gyro = new Pigeon2(0);
15        kinematics = new SwerveDriveKinematics(FrontLeft.place, FrontRight.place, BackLeft.place, BackRight.place);
16    }
17
18    public Command drive(Supplier<LinearVelocity> vx, Supplier<LinearVelocity> vy, Supplier<AngularVelocity> omega){
19        return run(() -> setState(kinematics.toSwerveModuleStates(new ChassisSpeeds(vx.get(), vy.get(), omega.get()))));
20    }
21
22    private void setState(SwerveModuleState[] states){
23        IntStream.range(0, 4).forEach(i -> modules.get(i).setState(states[i]));
24    }
25}
